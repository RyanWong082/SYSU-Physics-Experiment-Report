\section{实验目的}
\begin{enumerate}
  \item 学习并熟练掌握 Multisim 电路仿真软件的基本用法与工作流程。
  \item 使用 Multisim 对 RC、RL、RLC 串联电路进行频域仿真，得到幅频特性和相频特性曲线，并理解参数对谐振与相位的影响。
  \item 学会交流电桥（半桥与并联/串联型）的构成与调平衡方法，掌握在仿真环境下对未知电容与未知电感的参数测量。
\end{enumerate}

\section{仪器与用具}
\begin{itemize}
  \item 实验测控用计算机一台（安装 Multisim 与 LabVIEW 软件）。
\end{itemize}

\section{实验原理}
\subsection{Multisim 电路仿真软件简介}
Multisim 是基于工业标准 SPICE 引擎的电子线路仿真平台。其前身为 IIT 公司于 1988 年推出的 EWB（Electronics Workbench）。Multisim 集原理图绘制、器件库管理、虚拟仪器测量与数模混合仿真于一体，适合电路教学、方案验证与工程设计。2001 年起 EWB 更名为 Multisim；2005 年后并入 NI 生态，持续发展到多个版本。实验中以 Multisim~12.0 为例说明基本用法。

\subsection{Multisim~12.0 的界面与基本区块}
典型主界面包含：
\begin{itemize}
  \item \textbf{菜单与工具栏区}：文件管理、仿真控制、器件放置与虚拟仪器调用等。
  \item \textbf{项目/层级视图}：显示当前工程内各设计页与层级关系。
  \item \textbf{器件与仪器工具栏}：常用元件（电阻、电容、电感、二极管、三极管、运放等）与虚拟仪器（示波器 XSC、信号源 XFG、多用表 XMM、功率计等）。
  \item \textbf{电路设计工作区}：原理图绘制、连线与布局的主面板。
  \item \textbf{状态栏}：显示提示信息、仿真状态与光标坐标等。
\end{itemize}

\subsection{Multisim 的一般仿真流程}
\begin{enumerate}
  \item \textbf{新建设计文件}：\verb|File > New > Design|，或快捷键 \verb|Ctrl+N|。
  \item \textbf{放置元器件}：\verb|Place > Component| 打开元件库。库分为主数据库（Master Database）、用户数据库（User Database）与企业数据库（Corporate Database）。在对话框中按类别或检索关键词选择器件，点击 \texttt{OK} 放置到工作区。
  \item \textbf{放置虚拟仪器}：常用有信号发生器（XFG）、双踪示波器（XSC）、多用表（XMM）等。将仪器端口与电路节点正确连接。
  \item \textbf{参数设置与连线}：为电阻、电容、电感、源与仪器设置额定值、容差与量程。使用连线工具连接节点，注意接地端（\texttt{GND}）必须存在。
  \item \textbf{启动仿真}：点击“启动仿真”或 \verb|Simulate| 菜单运行。示波器可设置耦合方式、垂直/水平刻度、触发条件、扫描时间等。
  \item \textbf{数据读取与保存}：在仪器表头读数或在示波器上光标测量，必要时导出曲线数据；\verb|File > Save| 保存工程。
\end{enumerate}

\section{实验内容与步骤}
\subsection{任务一：学习 Multisim 用法并仿真 RLC 串联电路的频率响应}
\subsubsection*{电路与器件参数}
设置串联 RLC 电路，参数建议：
\[
R=\SI{1000}{\ohm},\qquad L=\SI{4.7}{mH},\qquad C=\SI{0.47}{\micro F}.
\]
在电路输入端接正弦信号源，幅度合适，频率进行扫频；在电容或电阻两端放置电压测量表头或示波器通道。

\subsubsection*{操作步骤}
\begin{enumerate}
  \item 在工作区放置 $R$、$L$、$C$ 串联器件及地端，连接成回路。输入端接 XFG，测量点并接 XMM 或示波器 XSC 通道。
  \item 设定 XFG 输出为正弦波，初始频率如 \SI{100}{Hz}，幅度满足后级不失真；在 \SI{100}{Hz}--\SI{100}{kHz} 范围内对数扫频或分点手动改变频率。
  \item 记录输出幅值 $|U|$ 与相位 $\varphi$ 随频率 $f$ 的变化，绘制幅频与相频特性。识别谐振频率 $f_0=\dfrac{1}{2\pi\sqrt{LC}}$ 及品质因数随 $R$ 的变化趋势。
  \item 改变为 RC 串联、RL 串联两种情形，重复测量并比较三者差异。
\end{enumerate}

\subsection{任务二：半桥式交流电桥的搭建与调平衡}
\subsubsection*{目的}
练习通过调节电桥臂阻值，使跨点 $A$、$B$ 电势差最小，实现电桥平衡；并在非平衡时观察输出电压与桥臂参数的关系。

\subsubsection*{步骤}
\begin{enumerate}
  \item 建立半桥电路，$R_1$、$R_3$ 取定值，$R_2$ 与 $R_4$ 可调。跨点 $A$、$B$ 接高内阻电压表或示波器差分测量。
  \item 保持 $R_1$、$R_3$ 不变，分别改变 $R_2$、$R_4$，使 $A$--$B$ 间电压差依次为 \SI{40}{mV}、\SI{30}{mV}、\SI{20}{mV}、\SI{10}{mV}、\SI{0}{mV}。记录每次对应的 $\Delta R$。
  \item 对比仿真读数与理论平衡条件，并讨论不平衡时输出电压与桥臂变化量的近似线性关系和灵敏度。
\end{enumerate}

\subsection{任务三：用交流电桥（电容臂）测量电容参数（仿真）}
\subsubsection*{测量模型与激励}
考虑实际电容存在损耗，等效为理想电容 $C_1$ 串联一内阻 $R_1$。信号采用正弦，频率 $f=\SI{10}{kHz}$，峰峰值 $V_{\mathrm{pp}}=\SI{5}{V}$。

\subsubsection*{调平衡与读数}
\begin{enumerate}
  \item 调整 $R_4$ 与 $R_1$ 的可调范围，使 $A$、$B$ 处电压差最小（约 \SI{1}{mV} 以内）。若无法达到完全平衡，可说明激励幅度、噪声、相位设置或参数初值不合适。
  \item 记录平衡时 $R_2$、$R_3$、$R_4$ 及标准电容 $C_2$ 的读数。根据平衡条件可得等效串联参数：
  \begin{equation}
    R_1=\frac{R_2}{R_4}\,R_3\,, \tag{G4.1}
  \end{equation}
  \begin{equation}
    C_1=\frac{R_4}{R_2}\,C_2\,. \tag{G4.2}
  \end{equation}
  损耗因数
  \begin{equation}
    D=\omega\,R_3\,C_2=\omega\,R_1\,C_1,\qquad \omega=2\pi f\,. \tag{G4.3}
  \end{equation}
\end{enumerate}

\subsection{任务四：用交流电桥（电感臂）测量电感参数（仿真）}
\subsubsection*{测量模型与激励}
被测电感等效为理想电感 $L_1$ 串联内阻 $R_1$。信号仍取 $f=\SI{10}{kHz}$，$V_{\mathrm{pp}}=\SI{5}{V}$。

\subsubsection*{调平衡与读数}
\begin{enumerate}
  \item 在电桥中接入标准电容 $C_1$ 与电阻网络，调节 $L_x$ 与 $R_4$ 的位置与数值，使 $A$、$B$ 间电压差最小（\SI{1}{mV} 左右）。
  \item 记录平衡读数 $R_2$、$R_3$、$R_4$ 与 $C_1$，按下式计算：
  \begin{equation}
    R_1=\frac{R_2}{R_4}\,R_3\,, \tag{G4.4}
  \end{equation}
  \begin{equation}
    L_1=C_1\,R_2\,R_3\,, \tag{G4.5}
  \end{equation}
  品质因数
  \begin{equation}
    Q=\omega\,C_1\,R_4\,. \tag{G4.6}
  \end{equation}
\end{enumerate}

\subsection{注意事项与误差讨论}
\begin{itemize}
  \item \textbf{地线与参考节点}：仿真电路必须放置地端，否则仪器读数与示波器触发可能异常。
  \item \textbf{仪器设置}：示波器的耦合方式、灵敏度、时间基与触发条件应与测量目标相匹配；信号源幅度不宜过大以免削顶。
  \item \textbf{平衡判据}：$A$、$B$ 间电压差小于 \SI{1}{mV} 可视作平衡。若噪声较大，可通过增大频率或延长平均时间改善。
  \item \textbf{参数耦合与灵敏度}：靠近平衡点时输出对桥臂微小变化更敏感；当两臂交换后重复测量，可减小系统误差并检验模型等效性。
  \item \textbf{数据处理}：建议记录扫频点的幅值与相位，绘制波特图；对比 RC、RL、RLC 三者的截止频率、相位拐点与谐振峰值。
\end{itemize}

\subsection*{E.\;利用 Multisim 软件仿真交流电桥测电容}
在本实验中，根据电桥接线图及表中给出的平衡条件，已知电桥的各项初始参数为：
\[
R_2 = 510\,\Omega,\qquad
R_3 = 10.85\,\Omega,\qquad
R_4 = 108.5\,\Omega,
\]
\[
C_2 = 0.47\,\mu\text{F},\qquad
f = 10\,\text{kHz},\qquad
\omega = 2\pi f.
\]

根据交流电桥的平衡条件，可分别计算待测电容的等效串联电阻、等效电容以及损耗因数：
\[
R_1 = \frac{R_2}{R_4}R_3,
\qquad
C_1 = \frac{R_4}{R_2}C_2,
\qquad
D = R_3 C_2 \omega.
\]

将测得数据代入上式可得本实验中被测电容的等效参数：
\[
R_1 = 51.0\,\Omega,\qquad
C_1 = 0.10\,\mu\text{F},\qquad
D = 0.32.
\]

由此可见，该电容的等效串联电阻较小，而损耗因数亦处于典型电解电容的正常范围之内。

\[
R_2 = 83.4\,\Omega,\qquad
R_3 = 510\,\Omega,\qquad
R_4 = 425.4\,\Omega,
\]
\[
C_1 = 0.47\,\mu\text{F},\qquad

根据交流电桥的平衡条件，可分别计算待测电感的等效电阻、电感量以及品质因数：

\section{思考题}
\begin{enumerate}
  \item 除 Multisim 外，请列举不少于 5 款常见电路仿真或 EDA 软件，并简单说明其特点或典型应用场景。\\[0.5em]
  \textbf{答：}
  \begin{itemize}
    \item \textbf{LTspice}：线性科技/ADI 提供的免费 SPICE 仿真软件，适合模拟电源、电放大器等线性电路的瞬态与频率响应分析。
    \item \textbf{PSpice / OrCAD}：业界常用的商用 SPICE 平台，与 PCB 设计流程集成良好，适合教学和工业级电路验证。
    \item \textbf{Proteus}：支持原理图 + 单片机程序联合仿真，可直接运行 MCU 固件，常用于嵌入式系统与单片机课程设计。
    \item \textbf{Altium Designer}：主打 PCB 设计与系统级 EDA，集原理图、PCB、信号完整性分析于一体，适合整机硬件开发。
    \item \textbf{Cadence Virtuoso}：面向集成电路（尤其是模拟 / 混合信号）的版图与电路设计平台，用于芯片级电路仿真与版图设计。
    \item \textbf{ADS（Advanced Design System）}：Keysight 的高频 / 射频仿真平台，适合微波电路、射频前端、天线和通信系统设计。
  \end{itemize}

  \item 推导直流电桥在半桥、全桥等不同拓扑下的输出电压与桥臂电阻偏差之间的关系式。\\[0.5em]
  \textbf{答：}
  设电源电压为 $U_s$，上、下臂电阻分别为 $R_1,R_2$ 与 $R_3,R_4$，输出端为中点 $A,B$。\\
  半桥或一般惠斯登电桥中：
  \[
    U_A = U_s\frac{R_2}{R_1+R_2},\qquad
    U_B = U_s\frac{R_4}{R_3+R_4},
  \]
  因此
  \[
    U_{AB} = U_A - U_B
            = U_s\!\left(
              \frac{R_2}{R_1+R_2}
              -\frac{R_4}{R_3+R_4}
            \right).
  \]
  若四臂名义值均为 $R_0$，其中
  \(
    R_2 = R_0 + \Delta R_2,\;
    R_4 = R_0 + \Delta R_4,
  \)
  且 $|\Delta R_i|\ll R_0$，在一阶近似下有
  \[
    U_{AB} \approx \frac{U_s}{4R_0}\left(\Delta R_2 - \Delta R_4\right),
  \]
  即输出电压与桥臂电阻的微小偏差成线性关系。\\
  对于对称全桥（四个桥臂均为可变电阻或应变片），可写成
  \[
    R_1 = R_0 + \Delta R_1,\;
    R_2 = R_0 + \Delta R_2,\;
    R_3 = R_0 + \Delta R_3,\;
    R_4 = R_0 + \Delta R_4,
  \]
  一阶线性化后
  \[
    U_{AB} \approx
    \frac{U_s}{4R_0}\left(
      \Delta R_1 - \Delta R_2 + \Delta R_4 - \Delta R_3
    \right),
  \]
  显示全桥可对多个桥臂的微小变化实现叠加放大，常用于高灵敏度传感测量。

  \item 结合工程实例，举出若干非平衡交流电桥在参数测试或在线监测中的应用。\\[0.5em]
  \textbf{答：}
  \begin{itemize}
    \item \textbf{电缆和绝缘测试}：Schering 电桥用于测量电力电缆、变压器绝缘的电容和介质损耗角，从而评估绝缘老化程度。
    \item \textbf{电感与线圈参数测量}：Maxwell、Hay 等交流电桥用于精确测量线圈电感量、等效串联电阻和品质因数，常用于变压器、电机绕组检测。
    \item \textbf{传感器测量}：电容式、感抗式传感器（如液位、湿度、压力传感器）可接入交流桥，利用桥路不平衡量反映被测物理量。
    \item \textbf{在线状态监测}：通过交流电桥实时监测某元件阻抗变化，如监控加热元件、电极、电化学传感器等的工作状态和老化趋势。
    \item \textbf{频率与滤波特性测试}：Wien 电桥等可用来标定信号源频率、测试有源滤波器或振荡器的幅频与相频特性。
  \end{itemize}

  \item 在交流电桥测电容、测电感实验中，为什么需要交换被测臂与标准臂的位置并重复测量？请从系统误差与模型等效性的角度分析。\\[0.5em]
  \textbf{答：}
  \begin{itemize}
    \item \textbf{减小系统误差}：励磁源、检流计（或检波器）、接线电阻及引线分布参数在电桥各臂上的影响并不完全相同。交换被测臂与标准臂后再次平衡，相当于对这些非对称误差取平均，可部分抵消系统误差。
    \item \textbf{检验等效模型}：被测元件通常用“理想元件 + 损耗”来等效（如 $C\!-\!R$ 串联或 $L\!-\!R$ 串联），但真实器件可能还含有寄生电感、电容等。交换位置后若两次测得参数接近，说明所采用的等效模型在本频率范围内是合理的。
    \item \textbf{提高结果可信度}：两次测量独立得到一组参数，如 $C_1^{(1)},\,C_1^{(2)}$ 或 $L_1^{(1)},\,L_1^{(2)}$，可以取平均值并估计差值作为不确定度，使最终给出的电容、电感和损耗因数 / 品质因数更可靠。
  \end{itemize}

\end{enumerate}
